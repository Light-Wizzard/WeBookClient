380QToolBar *BrowserWindow::createToolBar()
381{
382 QToolBar *navigationBar = new QToolBar(tr("Navigation"));
383 navigationBar->setMovable(false);
384 navigationBar->toggleViewAction()->setEnabled(false);
385
386 myHistoryBackAction = new QAction(this);
387 QList<QKeySequence> backShortcuts = QKeySequence::keyBindings(QKeySequence::Back);
388 for (auto it = backShortcuts.begin(); it != backShortcuts.end();)
389 {
390 // Chromium already handles navigate on backspace when appropriate.
391 if ((*it)[0] == Qt::Key_Backspace)
392 { it = backShortcuts.erase(it); }
393 else
394 { ++it; }
395 }
396 // For some reason Qt doesn't bind the dedicated Back key to Back.
397 backShortcuts.append(QKeySequence(Qt::Key_Back));
398 myHistoryBackAction->setShortcuts(backShortcuts);
399 myHistoryBackAction->setIconVisibleInMenu(false);
400 myHistoryBackAction->setIcon(QIcon(QStringLiteral(":go-previous.png")));
401 myHistoryBackAction->setToolTip(tr("Go back in history"));
402 connect(myHistoryBackAction, &QAction::triggered, [this]() { myTabWidget->triggerWebPageAction(QWebEnginePage::Back); });
403 navigationBar->addAction(myHistoryBackAction);
404
405 myHistoryForwardAction = new QAction(this);
406 QList<QKeySequence> fwdShortcuts = QKeySequence::keyBindings(QKeySequence::Forward);
407 for (auto it = fwdShortcuts.begin(); it != fwdShortcuts.end();)
408 {
409 if (((*it)[0] & Qt::Key_unknown) == Qt::Key_Backspace)
410 { it = fwdShortcuts.erase(it); }
411 else
412 { ++it; }
413 }
414 fwdShortcuts.append(QKeySequence(Qt::Key_Forward));
415 myHistoryForwardAction->setShortcuts(fwdShortcuts);
