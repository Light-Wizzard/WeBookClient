102void XbelTreeReader::readXBEL()
103{
104 Q_ASSERT(xml.isStartElement() && xml.name() == QLatin1String("xbel"));
105
106 while (xml.readNextStartElement())
107 {
108 if (xml.name() == QLatin1String("folder")) { readFolder(0); }
109 else if (xml.name() == QLatin1String("bookmark")) { readBookmark(0); }
110 else if (xml.name() == QLatin1String("separator")) { readSeparator(0); }
111 else { xml.skipCurrentElement(); }
