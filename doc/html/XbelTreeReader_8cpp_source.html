105void XbelTreeReader::readXBEL()
106{
107 Q_ASSERT(xml.isStartElement() && xml.name() == QLatin1String("xbel"));
108
109 while (xml.readNextStartElement())

112 else if (xml.name() == QLatin1String("bookmark")) { readBookmark(0); }
