84inline QString DownloadWidget::withUnit(qreal bytes)
85{
86 if (bytes < (1 << 10)) { return tr("%L1 B").arg(bytes); }
87 else if (bytes < (1 << 20)) { return tr("%L1 KiB").arg(bytes / (1 << 10), 0, 'f', 2); }
88 else if (bytes < (1 << 30)) { return tr("%L1 MiB").arg(bytes / (1 << 20), 0, 'f', 2); }
89 else { return tr("%L1 GiB").arg(bytes / (1 << 30), 0, 'f', 2); }
90}

95void DownloadWidget::updateWidget()
96{
97 qreal totalBytes = myDownload->totalBytes();
98 qreal receivedBytes = myDownload->receivedBytes();
99 qreal bytesPerSecond = receivedBytes / myTimeAdded.elapsed() * 1000;
100
101 auto state = myDownload->state();
102 switch (state)
103 {
104 case QWebEngineDownloadItem::DownloadRequested:
105 Q_UNREACHABLE();
106 break;
107 case QWebEngineDownloadItem::DownloadInProgress:
108 if (totalBytes >= 0)
109 {
110 progressBar->setValue(qRound(100 * receivedBytes / totalBytes));
111 progressBar->setDisabled(false);
112 progressBar->setFormat(tr("%p% - %1 of %2 downloaded - %3/s").arg(withUnit(receivedBytes), withUnit(totalBytes), withUnit(bytesPerSecond)));
113 }
114 else
115 {
116 progressBar->setValue(0);
117 progressBar->setDisabled(false);
118 progressBar->setFormat(tr("unknown size - %1 downloaded - %2/s").arg(withUnit(receivedBytes), withUnit(bytesPerSecond)));
119 }
120 break;
121 case QWebEngineDownloadItem::DownloadCompleted:
122 progressBar->setValue(100);
123 progressBar->setDisabled(true);
124 progressBar->setFormat(tr("completed - %1 downloaded - %2/s").arg(withUnit(receivedBytes), withUnit(bytesPerSecond)));
125 break;
126 case QWebEngineDownloadItem::DownloadCancelled:
127 progressBar->setValue(0);
128 progressBar->setDisabled(true);
129 progressBar->setFormat(tr("cancelled - %1 downloaded - %2/s").arg(withUnit(receivedBytes), withUnit(bytesPerSecond)));
130 break;
131 case QWebEngineDownloadItem::DownloadInterrupted:
132 progressBar->setValue(0);
133 progressBar->setDisabled(true);
134 progressBar->setFormat(tr("interrupted: %1").arg(myDownload->interruptReasonString()));
135 break;
136 }
137
138 if (state == QWebEngineDownloadItem::DownloadInProgress)
139 {
140 static QIcon cancelIcon(QStringLiteral(":process-stop.png"));
141 buttonCancel->setIcon(cancelIcon);
142 buttonCancel->setToolTip(tr("Stop downloading"));
143 }
144 else
145 {
146 static QIcon removeIcon(QStringLiteral(":edit-clear.png"));
147 buttonCancel->setIcon(removeIcon);
148 buttonCancel->setToolTip(tr("Remove from list"));
149 }
150}
