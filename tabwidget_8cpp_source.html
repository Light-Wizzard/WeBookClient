136void TabWidget::handleContextMenuRequested(const QPoint &pos)
137{
138 QMenu menu;
139 menu.addAction(tr("New &Tab"), this, &TabWidget::createTab, QKeySequence::AddTab);
140 int index = tabBar()->tabAt(pos);
141 if (index != -1)
142 {
143 QAction *action = menu.addAction(tr("Clone Tab"));
144 connect(action, &QAction::triggered, this, [this,index]() { cloneTab(index); });
145 menu.addSeparator();
146 action = menu.addAction(tr("&Close Tab"));
147 action->setShortcut(QKeySequence::Close);
148 connect(action, &QAction::triggered, this, [this,index]() { closeTab(index); });
149 action = menu.addAction(tr("Close &Other Tabs"));
150 connect(action, &QAction::triggered, this, [this,index]() { closeOtherTabs(index); });
151 menu.addSeparator();
152 action = menu.addAction(tr("Reload Tab"));
153 action->setShortcut(QKeySequence::Refresh);
154 connect(action, &QAction::triggered, this, [this,index]() { reloadTab(index); });
155 }
156 else
157 {
158 menu.addSeparator();
159 }
160 menu.addAction(tr("Reload All Tabs"), this, &TabWidget::reloadAllTabs);
161 menu.exec(QCursor::pos());
162}
