131{
132 if (myTabWidget->count() > 1)
133 {
134 int theOpenFeatureTabs = 0;
135 if (myTabWidget->getBookmarkTab() != -1) { theOpenFeatureTabs++; }

138 theOpenFeatureTabs = theOpenFeatureTabs - myTabWidget->count();
139 if (theOpenFeatureTabs > 1)
140 {
