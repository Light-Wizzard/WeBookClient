172QWebEngineView *WebView::createWindow(QWebEnginePage::WebWindowType type)
173{
174 BrowserWindow *mainWindow = qobject_cast<BrowserWindow*>(window());
175 if (!mainWindow) { return nullptr; }
176
177 switch (type)
178 {
179 case QWebEnginePage::WebBrowserTab: { return mainWindow->tabWidget()->createTab(); }
180 case QWebEnginePage::WebBrowserBackgroundTab: { return mainWindow->tabWidget()->createBackgroundTab(); }
181 case QWebEnginePage::WebBrowserWindow: { return mainWindow->browser()->createWindow()->currentTab(); }
182 case QWebEnginePage::WebDialog:
183 {
184 WebPopupWindow *popup = new WebPopupWindow(page()->profile());
185 connect(popup->view(), &WebView::devToolsRequested, this, &WebView::devToolsRequested);
186 return popup->view();
187 }
188 }
189 return nullptr;
190}

197{

199 const QList<QAction *> actions = menu->actions();
200 auto inspectElement = std::find(actions.cbegin(), actions.cend(), page()->action(QWebEnginePage::InspectElement));
201 if (inspectElement == actions.cend())
202 {
203 auto viewSource = std::find(actions.cbegin(), actions.cend(), page()->action(QWebEnginePage::ViewSource));
204 if (viewSource == actions.cend()) { menu->addSeparator(); }
205
206 QAction *action = new QAction(menu);
207 action->setText("Open inspector in new window");
208 connect(action, &QAction::triggered, [this]() { emit devToolsRequested(page()); });
209
210 QAction *before(inspectElement == actions.cend() ? nullptr : *inspectElement);
211 menu->insertAction(before, action);
212 }
213 else
214 {
215 (*inspectElement)->setText(tr("Inspect element"));
216 }
217 menu->popup(event->globalPos());
