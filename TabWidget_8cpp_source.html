127void TabWidget::handleContextMenuRequested(const QPoint &pos)
128{
129 QMenu menu;
130 menu.addAction(tr("New &Tab"), this, &TabWidget::createTab, QKeySequence::AddTab);
131 int index = tabBar()->tabAt(pos);
132 if (index != -1)
133 {
134 QAction *action = menu.addAction(tr("Clone Tab"));
135 connect(action, &QAction::triggered, this, [this,index]() { cloneTab(index); });
136 menu.addSeparator();
137 action = menu.addAction(tr("&Close Tab"));
138 action->setShortcut(QKeySequence::Close);
139 connect(action, &QAction::triggered, this, [this,index]() { closeTab(index); });
140 action = menu.addAction(tr("Close &Other Tabs"));
141 connect(action, &QAction::triggered, this, [this,index]() { closeOtherTabs(index); });
142 menu.addSeparator();
143 action = menu.addAction(tr("Reload Tab"));
144 action->setShortcut(QKeySequence::Refresh);
145 connect(action, &QAction::triggered, this, [this,index]() { reloadTab(index); });
146 }
147 else
148 {
149 menu.addSeparator();
150 }
151 menu.addAction(tr("Reload All Tabs"), this, &TabWidget::reloadAllTabs);
152 menu.exec(QCursor::pos());
153}

202 int index = indexOf(webView);

204 if (currentIndex() == index) { emit favIconChanged(icon); }
205 });

207 connect(webPage, &QWebEnginePage::windowCloseRequested, [this, webView]()
208 {
209 int index = indexOf(webView);

211 });

473void TabWidget::closeTab(int thisIndex)
474{
475 if (myBookmarkTab == thisIndex) { setTabVisible(thisIndex, false); return; }
476 if (myHelpTab == thisIndex) { setTabVisible(thisIndex, false); return; }
477 if (myDownloadTab == thisIndex) { setTabVisible(thisIndex, false); return; }
478
479 if (WebView *view = webView(thisIndex))
480 {
481 bool hasFocus = view->hasFocus();
482 removeTab(thisIndex);
483 if (hasFocus && count() > 0) { currentWebView()->setFocus(); }
484 if (count() == 0) { createTab(); }
485 view->deleteLater();
486 }
487}
